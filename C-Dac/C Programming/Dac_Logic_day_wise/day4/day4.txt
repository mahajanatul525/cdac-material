Q1. Give choice to user
    a.	square
    b.	Cube
    c.	exit
    as per user choice display result if user enter invalid choice give appropriate 
    message.

ans:

#include<stdio.h>
void main()
{
	char ch;
	int n;
	printf("\nEnter choice : a. square\t b. cube\t c. exit\n");
	scanf("%c",&ch);
	if(ch=='a' || ch=='A' || ch=='b' || ch=='B')
	{
		 printf("\nEnter no. = ");
		 scanf("%d",&n);
	}
	switch(ch)
	{
		case 'a':
		case 'A': printf("Square = %d",n*n);
			  break;

		case 'b':
		case 'B': printf("Cube = %d",n*n*n);
			  break;

		case 'c':
		case 'C': exit(0);

		default: printf("You have entered invalid choice");
	}

}

Recursive function :

Q1. Find factorial of a number

ans:

#include<stdio.h>
int rec(int);
void main()
{
	int no,f;
	printf("\nEnter number = ");
	scanf("%d",&no);
	f=rec(no);
	printf("Factorial of number = %d",f);
}

int rec(int x)
{
	int f;
	if(x==1)
		return 1;
	else
	{
		f=x*rec(x-1);
		return f;
	}
}

{

Q3. Find Fibonacci series

ans:

#include<stdio.h>
void fib(int);
void main()
{
	int t;
	printf("\nEnter term = ");
	scanf("%d",&t);
	if(t==0)
		printf("Enter valid term");
	else if(t==1)
		printf("0");
	else if(t==2)
		printf("0\t1");
	else
	{
		printf("0\t1");
		fib(t);
	}
}

void fib(int n)
{
	static int x=0, y=1;
	int c;
	c = x + y;
	printf("\t%d",c);
	x=y;
	y=c;
	n--;
	if(n>2)
	fib(n);
}
